  ```python
  from abc import ABC, abstractmethod
  from typing import List, Optional
  
  class NotificationChannel(ABC):
      @abstractmethod
      async def send(self, message: str, subject: Optional[str] = None) -> bool:
          pass
  
  class TelegramChannel(NotificationChannel):
      async def send(self, message: str, subject: Optional[str] = None) -> bool:
          # TODO: implement Telegram API call
          pass
  
  class EmailChannel(NotificationChannel):
      async def send(self, message: str, subject: Optional[str] = None) -> bool:
          # TODO: implement SMTP/email API call
          pass
  
  class NotificationService:
      def __init__(self, channels: List[NotificationChannel]):
          self.channels = channels
  
      async def notify(self, message: str, subject: Optional[str] = None):
          for channel in self.channels:
              await channel.send(message, subject)
  ```
- `backend/app/services/scheduler_service.py:1-100` — `SchedulerService` — فایل جدید — زمان‌بندی و یادآوری وظایف
  ```python
  from apscheduler.schedulers.asyncio import AsyncIOScheduler
  from datetime import datetime, timedelta
  
  class SchedulerService:
      def __init__(self, notification_service: NotificationService):
          self.scheduler = AsyncIOScheduler()
          self.notification_service = notification_service
  
      def schedule_reminder(self, reminder_time: datetime, message: str):
          self.scheduler.add_job(
              self.notification_service.notify,
              'date',
              run_date=reminder_time,
              args=[message, 'Reminder']
          )
  ```
