## 🔍 Context و وضعیت فعلی
این تسک شامل پیاده‌سازی منطق تشخیص `error_rate > 1%` و فعال‌سازی هشدار (alert) در سیستم مانیتورینگ پروژه است. طبق درخواست کاربر، کد باید بررسی کند که اگر `error_rate` از آستانهٔ `error_rate_threshold` یعنی ۱٪ عبور کرد، یک هشدار (`send_alert`) فعال شود. این مرحله صرفاً منطق تشخیص و فعال‌سازی هشدار را پوشش می‌دهد — اتصال به کانال‌های اطلاع‌رسانی (مثل Telegram یا Email از طریق `notification_service`) خارج از scope این مرحله است.

# backend/app/services/error_rate_monitor.py
from __future__ import annotations
import logging
from datetime import datetime
from typing import Optional, Dict, Any

logger = logging.getLogger(__name__)

ERROR_RATE_THRESHOLD = 1.0  # درصد


def calculate_error_rate(total_requests: int, failed_requests: int) -> float:
    """محاسبهٔ نرخ خطا به درصد."""
    if total_requests == 0:
        return 0.0
    return (failed_requests / total_requests) * 100


def check_error_rate_and_alert(
    total_requests: int,
    failed_requests: int,
    project_id: Optional[str] = None,
) -> Optional[Dict[str, Any]]:
    """بررسی error_rate و تولید alert اگر از آستانه عبور کرد.
    
    خروجی: dict هشدار یا None اگر زیر آستانه باشد.
    اتصال به کانال‌های اطلاع‌رسانی در مرحلهٔ بعد انجام می‌شود.
    """
    error_rate = calculate_error_rate(total_requests, failed_requests)
    
    if error_rate > ERROR_RATE_THRESHOLD:
        alert = {
            "event": "error_rate_alert",
            "error_rate": round(error_rate, 4),
            "threshold": ERROR_RATE_THRESHOLD,
            "total_requests": total_requests,
            "failed_requests": failed_requests,
            "triggered_at": datetime.utcnow().isoformat(),
            "project_id": project_id,
        }
        logger.warning(
            f"error_rate_alert: error_rate={error_rate:.2f}% > threshold={ERROR_RATE_THRESHOLD}%"
            f" | project_id={project_id}"
        )
        return alert
    
    return None
```

class SlackChannel(NotificationChannel):
    name = "slack"

    def __init__(self, webhook_url: Optional[str]):
        self.webhook_url = (webhook_url or os.environ.get("SLACK_WEBHOOK_URL", "")).strip()

    def is_configured(self) -> bool:
        return bool(self.webhook_url)

    async def send(
        self, message: str, *, subject: Optional[str] = None,
        silent: bool = False, reply_markup: Optional[Dict[str, Any]] = None,
    ) -> Dict[str, Any]:
        if not self.is_configured():
            return {"ok": False, "channel": self.name, "error": "SLACK_WEBHOOK_URL خالی است"}
        text = f"*{subject}*\n{message}" if subject else message
        payload = {"text": text[:3000]}
        try:
            timeout = aiohttp.ClientTimeout(total=15)
            async with aiohttp.ClientSession(timeout=timeout) as session:
                async with session.post(self.webhook_url, json=payload) as r:
                    if r.status != 200:
                        body = await r.text()
                        return {"ok": False, "channel": self.name, "error": f"HTTP {r.status}: {body[:200]}"}
                    return {"ok": True, "channel": self.name}
        except Exception as e:
            return {"ok": False, "channel": self.name, "error": str(e)[:300]}

async def send_alert(
    event: str,
    message: str,
    *,
    subject: Optional[str] = None,
    priority: Optional[str] = None,
    project_name: Optional[str] = None,
    silent: Optional[bool] = None,
    extra_buttons: Optional[List[Dict[str, str]]] = None,
) -> List[Dict[str, Any]]:
    """ارسال هشدار به همه کانال‌های فعال بر اساس prefs."""
    prefs = _read_prefs()
    # بررسی فعال بودن event
    if not prefs.get("events", {}).get(event, True):
        logger.debug(f"send_alert: event '{event}' disabled in prefs")
        return []
    # تعیین silent بر اساس prefs اگر override نشده
    if silent is None:
        silent = not prefs.get("sound", {}).get(event, False)
    # ساخت hashtags
    hashtags = ""
    if prefs.get("include_hashtags"):
        hashtags = "\n" + build_hashtags(event, project_name=project_name, priority=priority)
    full_message = message + hashtags
    # inline keyboard
    reply_markup = None
    if prefs.get("include_inline_buttons"):
        base_url = prefs.get("app_base_url", "")
        reply_markup = build_inline_keyboard(base_url, event, extra_buttons=extra_buttons)
    # instantiate کانال‌های فعال
    channels: List[NotificationChannel] = []
    ch_prefs = prefs.get("channels", {})
    if ch_prefs.get("telegram", {}).get("enabled"):
        channels.append(TelegramChannel(
            os.environ.get("TELEGRAM_BOT_TOKEN"),
            os.environ.get("TELEGRAM_CHAT_ID"),
        ))
    if ch_prefs.get("email", {}).get("enabled"):
        channels.append(EmailChannel())  # از env vars می‌خواند
    if ch_prefs.get("slack", {}).get("enabled"):
        channels.append(SlackChannel(None))  # از SLACK_WEBHOOK_URL می‌خواند
    if ch_prefs.get("webhook", {}).get("enabled"):
        channels.append(WebhookChannel(None))  # از NOTIFICATION_WEBHOOK_URL می‌خواند
    if not channels:
        logger.debug("send_alert: no channels configured/enabled")
        return []
    # ارسال concurrent
    tasks = [
        ch.send(full_message, subject=subject, silent=silent,
                reply_markup=reply_markup if ch.name == "telegram" else None)
        for ch in channels if ch.is_configured()
    ]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    out = []
    for ch, res in zip([c for c in channels if c.is_configured()], results):
        if isinstance(res, Exception):
            logger.error(f"send_alert channel={ch.name} exception: {res}")
            out.append({"ok": False, "channel": ch.name, "error": str(res)})
        else:
            if not res.get("ok"):
                logger.warning(f"send_alert channel={ch.name} failed: {res.get('error')}")
            out.append(res)
    return out
```
